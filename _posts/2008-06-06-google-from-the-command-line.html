#!/bin/bash
query="$@"
if [ -z "$query" ]; then
  echo "Usage:   google \"
  echo "You must specify a search query"
  exit -1
fi
query=`echo $query | sed -e s/'\ '/%20/g`
w3m -dump -no-cookie -cols 200 http://www.google.com/search?q=${query} | grep Cached | cut -f 1 -d' '
